1. Models (ORM)

Model nomi: Books

  • nomi (book name)
  • yili (year of publication)
  • muallif (author)
  • url (for redirect/detail)
from django.db import models 

class Book(models.Model):
nomi = models.CharField(max_length=200)
yili = models.IntegerField()
muallif = models.CharField(max_length=200)
url = models.URLField(blank=True)

def __str__(self):
    return self.nomi

2. Views

Sahifalar:

  • Base: Bosh layout (navbar, footer)
  • Home: Asosiy sahifa
  • Books: Kitoblar ro‘yxati
  • Detail B: Kitobning batafsil sahifasi
  • Edit Books: Kitobni tahrirlash
  • Create B: Yangi kitob qo‘shish
from django.shortcuts import render, get_object_or_404, redirect
from .models import Book
from .forms import BookForm

# Home sahifasi
def home(request):
    return render(request, 'home.html')

# Kitoblar ro'yxati
def book_list(request):
    books = Book.objects.all()
    return render(request, 'books.html', {'books': books})

# Kitob batafsil
def book_detail(request, id):
    book = get_object_or_404(Book, id=id)
    return render(request, 'book_detail.html', {'book': book})

# Kitob qo'shish
def create_book(request):
    if request.method == 'POST':
        form = BookForm(request.POST)
        if form.is_valid():
            form.save()
            return redirect('books')
    else:
        form = BookForm()
    return render(request, 'create_book.html', {'form': form})

# Kitobni tahrirlash
def edit_book(request, id):
    book = get_object_or_404(Book, id=id)
    if request.method == 'POST':
        form = BookForm(request.POST, instance=book)
        if form.is_valid():
            form.save()
            return redirect('book_detail', id=book.id)
    else:
        form = BookForm(instance=book)
    return render(request, 'edit_book.html', {'form': form, 'book': book})

# Kitobni o'chirish
def delete_book(request, id):
    book = get_object_or_404(Book, id=id)
    if request.method == 'POST':
        book.delete()
        return redirect('books')
    return render(request, 'delete_book.html', {'book': book})

2. Foydalanuvchi autentifikatsiyasi va boshqa funksiyalar (views.py)

views.py fayli qatorini yangilang. Bu yerda:

  • book_list: Qidiruv (search) va pagination qo’llanilgan.
  • register: Ro’yxatdan o’tish funksiyasi.
  • login_required dekoratori yordamida ba’zi sahifalarda avtorizatsiya tekshirilishi mumkin (masalan, kitob qo’shish va tahrir qilish).
from django.shortcuts import render, get_object_or_404, redirect
from django.core.paginator import Paginator
from django.contrib.auth.forms import UserCreationForm
from django.contrib.auth import login
from django.contrib.auth.decorators import login_required
from .models import Book
from .forms import BookForm

# Bosh sahifa
def home(request):
    return render(request, 'home.html')

# Kitoblar ro'yxati - qidiruv va pagination qo'llanilgan
def book_list(request):
    query = request.GET.get('q', '')
    books = Book.objects.all()
    if query:
        books = books.filter(nomi__icontains=query)
    paginator = Paginator(books, 5)  # Sahifada 5 ta kitob
    page = request.GET.get('page')
    books_page = paginator.get_page(page)
    context = {'books': books_page, 'query': query}
    return render(request, 'books.html', context)

# Kitob batafsil
def book_detail(request, id):
    book = get_object_or_404(Book, id=id)
    return render(request, 'book_detail.html', {'book': book})

# Yangi kitob qo'shish (faqat avtorizatsiyadan o’tgan foydalanuvchilar uchun)
@login_required
def create_book(request):
    if request.method == 'POST':
        form = BookForm(request.POST)
        if form.is_valid():
            form.save()
            return redirect('books')
    else:
        form = BookForm()
    return render(request, 'create_book.html', {'form': form})

# Kitobni tahrirlash (faqat avtorizatsiyadan o’tgan foydalanuvchilar uchun)
@login_required
def edit_book(request, id):
    book = get_object_or_404(Book, id=id)
    if request.method == 'POST':
        form = BookForm(request.POST, instance=book)
        if form.is_valid():
            form.save()
            return redirect('book_detail', id=book.id)
    else:
        form = BookForm(instance=book)
    return render(request, 'edit_book.html', {'form': form, 'book': book})

# Kitobni o'chirish (faqat avtorizatsiyadan o’tgan foydalanuvchilar uchun)
@login_required
def delete_book(request, id):
    book = get_object_or_404(Book, id=id)
    if request.method == 'POST':
        book.delete()
        return redirect('books')
    return render(request, 'delete_book.html', {'book': book})

# Ro'yxatdan o'tish
def register(request):
    if request.method == 'POST':
        form = UserCreationForm(request.POST)
        if form.is_valid():
            user = form.save()
            login(request, user)
            return redirect('home')
    else:
        form = UserCreationForm()
    return render(request, 'register.html', {'form': form})

3. URL'lar (urls.py)

urls.py faylida yangi marshrutlarni qo’shib, Django’ning autentifikatsiya tizimi uchun login va logout URL’larini ham ko’rsatamiz. (Eslatma: Django’ning LoginView va LogoutView dan foydalanamiz.)

from django.urls import path
from django.contrib.auth import views as auth_views
from . import views

urlpatterns = [
    path('', views.home, name='home'),
    path('books/', views.book_list, name='books'),
    path('book-detail/<int:id>/', views.book_detail, name='book_detail'),
    path('book-edit/<int:id>/', views.edit_book, name='edit_book'),
    path('create/', views.create_book, name='create_book'),
    path('delete-book/<int:id>/', views.delete_book, name='delete_book'),
    
    # Avtorizatsiya uchun URL'lar
    path('register/', views.register, name='register'),
    path('login/', auth_views.LoginView.as_view(template_name='login.html'), name='login'),
    path('logout/', auth_views.LogoutView.as_view(next_page='home'), name='logout'),
]

from django.shortcuts import render, get_object_or_404, redirect

from django.contrib.auth.forms import UserCreationForm

    if search_query:
        books = books.filter(title__icontains=search_query)
